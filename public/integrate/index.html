The contract

POST https://<app-domain>/<helper_path>      default helper_path: srvmgr/helper.php
Content-Type: application/json

Signed auth (preferred — the secret never travels on the wire):
X-Srvmgr-Timestamp: <unix seconds>           reject if > 300s old
X-Srvmgr-Nonce:     <random hex>             single-use
X-Srvmgr-Signature: v1=base64url(hmac_sha256(secret, canonical))
  canonical = "{ts}\n{nonce}\n{method}\n{path}\n{sha256(rawBody)}"
Legacy fallback:
X-Srvmgr-Token: <shared secret>              must equal the app's helper_token

Request:   { "action": "health", ...params }
Success:   200  { "ok": true,  "data": { ... } }
Auth fail: 401  { "ok": false, "error": "unauthorized" }
Unknown:   400  { "ok": false, "error": "unknown action" }
The shared secret is the HMAC key and is never sent in a header — only the signature is, so captured requests can't be replayed or forged. Verify signatures with a constant-time compare, serve over HTTPS, and never echo the token or DB credentials.

Standard actions

actionparamsdata to return
health required— { status:"ok"|"degraded", db:bool, time:ISO8601 }
stats—app counters, e.g. { users, sessions }
logs{ lines:200 }{ entries:[ { ip, method, path, status, bytes, time } ] } — feeds the traffic map
version—{ version:"<sha|semver>" }
clear_cache—{ cleared:true }
maintenance{ on:bool }{ maintenance:bool }
migrate destructive{ confirm:true } { migrated:true, applied:[] }

health.status of ok/healthy shows healthy; anything else shows degraded. Unknown actions must return HTTP 400.

Reference implementation

Copy to srvmgr/helper.php, set SRVMGR_HELPER_TOKEN, wire app_db(). Full file: helper.sample.php.txt

<?php
declare(strict_types=1);
header('Content-Type: application/json');

$TOKEN = getenv('SRVMGR_HELPER_TOKEN') ?: '';
$given = $_SERVER['HTTP_X_SRVMGR_TOKEN'] ?? '';
if ($TOKEN === '' || !hash_equals($TOKEN, $given)) {
    http_response_code(401);
    exit(json_encode(['ok'=>false,'error'=>'unauthorized']));
}
$req = json_decode(file_get_contents('php://input') ?: '[]', true) ?: [];
$reply = fn($d,$ok=true,$c=200)=>[http_response_code($c),
    print(json_encode(['ok'=>$ok,'data'=>$d]))][0];

switch ($req['action'] ?? '') {
  case 'health':
    $db = false;
    try { /* $pdo = your PDO; */ $db = true; } catch (Throwable $e) { $db = false; }
    $reply(['status'=>$db?'ok':'degraded','db'=>$db,'time'=>date('c')]); break;
  case 'stats':   $reply(['users'=>0]); break;
  default:        $reply(['error'=>'unknown action'], false, 400);
}
// routes/srvmgr.js  (mount at /srvmgr/helper)
const crypto = require('crypto');
module.exports = (req, res) => {
  const token = process.env.SRVMGR_HELPER_TOKEN || '';
  const given = req.get('X-Srvmgr-Token') || '';
  const a = Buffer.from(token), b = Buffer.from(given);
  if (!token || a.length !== b.length || !crypto.timingSafeEqual(a, b))
    return res.status(401).json({ ok:false, error:'unauthorized' });

  const reply = (data, ok=true, code=200) => res.status(code).json({ ok, data });
  switch ((req.body && req.body.action) || '') {
    case 'health': {
      let db = false;
      try { /* await pool.query('SELECT 1'); */ db = true; } catch { db = false; }
      return reply({ status: db?'ok':'degraded', db, time: new Date().toISOString() });
    }
    case 'stats': return reply({ users: 0 });
    default:      return reply({ error:'unknown action' }, false, 400);
  }
};
# flask route mounted at /srvmgr/helper
import os, hmac, datetime
from flask import request, jsonify

def srvmgr_helper():
    token = os.environ.get("SRVMGR_HELPER_TOKEN", "")
    given = request.headers.get("X-Srvmgr-Token", "")
    if not token or not hmac.compare_digest(token, given):
        return jsonify(ok=False, error="unauthorized"), 401

    def reply(data, ok=True, code=200):
        return jsonify(ok=ok, data=data), code

    action = (request.get_json(silent=True) or {}).get("action", "")
    if action == "health":
        try:  # db.session.execute("SELECT 1")
            db = True
        except Exception:
            db = False
        return reply({"status": "ok" if db else "degraded", "db": db,
                      "time": datetime.datetime.utcnow().isoformat() + "Z"})
    if action == "stats":
        return reply({"users": 0})
    return reply({"error": "unknown action"}, False, 400)
